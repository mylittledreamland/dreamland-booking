// ============================================================ // MY LITTLE DREAMLAND — Booking API v2 (Apps Script) // ============================================================ // Deploy: Extensions > Apps Script > Deploy > New deployment // Type: Web App | Execute as: Me | Access: Anyone // ============================================================ const CONFIG = { SHEET_ID: "1lRXe8Tt02mNZ33u6wIaoWNbxOz2CtAlwTbeGIemVnFE", OPEN_TIME: "08:30", CLOSE_TIME: "20:00", SLOT_DURATION: 15, // minutes between entry slots PLAY_DURATION: 60, // minutes each session lasts MAX_CAPACITY: 5, // max people per 15-min entry slot ALMOST_FULL_THRESHOLD: 2, SHEET_BOOKINGS: "Bookings", SHEET_CAPACITY: "Capacity", }; // ============================================================ // ROUTING // ============================================================ function doGet(e) { const action = e.parameter.action || "getSlots"; if (action === "getSlots") { const date = e.parameter.date; if (!date) return jsonError("Missing parameter: date"); return jsonResponse(getSlotsForDate(date)); } if (action === "createBooking") { const data = { action: "createBooking", date: e.parameter.date, slot: e.parameter.slot, parentName: e.parameter.parentName, parentEmail: e.parameter.parentEmail, parentPhone: e.parameter.parentPhone || "", numChildren: e.parameter.numChildren, photoConsent: e.parameter.photoConsent || "no", waiverSigned: e.parameter.waiverSigned || "no", notes: e.parameter.notes || "", }; return jsonResponse(createBooking(data)); } return jsonError("Invalid action"); } function doPost(e) { try { let body; if (e.parameter && e.parameter.data) { body = JSON.parse(e.parameter.data); } else if (e.postData && e.postData.contents) { body = JSON.parse(e.postData.contents); } else { return jsonError("Could not read request body"); } const action = body.action || "createBooking"; if (action === "createBooking") return jsonResponse(createBooking(body)); return jsonError("Invalid action"); } catch (err) { return jsonError("Invalid request: " + err.message); } } // ============================================================ // GET SLOTS // ============================================================ function getSlotsForDate(date) { const groupedSlots = generateSlots(); const booked = getBookedCountsByDate(date); const customCapacity = getCustomCapacity(date); const now = new Date(); const todayStr = Utilities.formatDate(now, "Australia/Melbourne", "yyyy-MM-dd"); const currentMinutes = parseInt(Utilities.formatDate(now, "Australia/Melbourne", "HH")) * 60 + parseInt(Utilities.formatDate(now, "Australia/Melbourne", "mm")); return { date: date, slotDuration: CONFIG.SLOT_DURATION, playDuration: CONFIG.PLAY_DURATION, groupedSlots: groupedSlots.map(group => { return { hourLabel: group.hourLabel, subSlots: group.subSlots.map((slot) => { const capacity = customCapacity[slot.key] ?? CONFIG.MAX_CAPACITY; const bookedCount = booked[slot.key] || 0; const remaining = Math.max(0, capacity - bookedCount); let isPast = false; if (date === todayStr) { const [h, m] = slot.start.split(":").map(Number); isPast = h * 60 + m <= currentMinutes; } return { key: slot.key, start: slot.start, end: slot.end, capacity, booked: bookedCount, remaining, status: isPast ? "past" : remaining === 0 ? "full" : remaining <= CONFIG.ALMOST_FULL_THRESHOLD ? "almostFull" : "available", }; }) }; }) }; } // ============================================================ // CREATE BOOKING // ============================================================ function createBooking(data) { const required = ["date", "slot", "parentName", "parentEmail", "numChildren"]; for (const field of required) { if (!data[field]) return { success: false, error: "Missing required field: " + field }; } const numChildren = parseInt(data.numChildren); if (isNaN(numChildren) || numChildren < 1 || numChildren > 10) { return { success: false, error: "Invalid number of children (1-10)" }; } const slotsData = getSlotsForDate(data.date); let slot = null; // Tìm slot trong mảng phân cấp giờ for (const group of slotsData.groupedSlots) { const found = group.subSlots.find(s => s.key === data.slot); if (found) { slot = found; break; } } if (!slot) return { success: false, error: "Session not found" }; if (slot.status === "past") return { success: false, error: "This session has already passed" }; if (slot.remaining < numChildren) { return { success: false, error: "Not enough spots. Remaining: " + slot.remaining + " spots", remaining: slot.remaining, }; } const ss = SpreadsheetApp.openById(CONFIG.SHEET_ID); const sheet = ss.getSheetByName(CONFIG.SHEET_BOOKINGS); const bookingId = generateId(); const timestamp = new Date(); const expiryTime = new Date(timestamp.getTime() + 15 * 60 * 1000); const expiryStr = Utilities.formatDate(expiryTime, "Australia/Melbourne", "yyyy-MM-dd HH:mm:ss"); sheet.appendRow([ bookingId, Utilities.formatDate(timestamp, "Australia/Melbourne", "yyyy-MM-dd HH:mm:ss"), data.date, data.slot, slot.start, slot.end, data.parentName, data.parentEmail, data.parentPhone || "", numChildren, data.waiverSigned || "no", data.photoConsent || "no", data.notes || "", "confirmed", expiryStr, ]); try { sendConfirmationEmail(data, bookingId, slot, expiryStr); } catch (emailErr) { console.log("Email error:", emailErr.message); } return { success: true, bookingId, message: "Booking confirmed!", booking: { id: bookingId, date: data.date, slot: data.slot, entryTime: slot.start, exitTime: slot.end, numChildren, parentName: data.parentName, holdExpiry: expiryStr, }, }; } // ============================================================ // HELPERS // ============================================================ function generateSlots() { const groups = []; const [openH, openM] = CONFIG.OPEN_TIME.split(":").map(Number); const [closeH, closeM] = CONFIG.CLOSE_TIME.split(":").map(Number); const openTotal = openH * 60 + openM; const closeTotal = closeH * 60 + closeM; let current = openTotal; let currentGroupHour = openH; let currentSubSlots = []; while (current + CONFIG.PLAY_DURATION <= closeTotal) { const h = Math.floor(current / 60); // Nếu qua giờ mới (vd từ 8 sang 9) và đã có slot bên trong, thì lưu lại thành 1 nhóm if (h !== currentGroupHour && currentSubSlots.length > 0) { const ampm = currentGroupHour >= 12 ? 'PM' : 'AM'; const displayHour = currentGroupHour > 12 ? currentGroupHour - 12 : (currentGroupHour === 0 ? 12 : currentGroupHour); groups.push({ hourLabel: `${displayHour}:00 ${ampm}`, subSlots: currentSubSlots }); currentGroupHour = h; currentSubSlots = []; } else if (h !== currentGroupHour && currentSubSlots.length === 0) { currentGroupHour = h; } const start = minutesToTime(current); const end = minutesToTime(current + CONFIG.PLAY_DURATION); currentSubSlots.push({ key: `${start}-${end}`, start: start, end: end }); current += CONFIG.SLOT_DURATION; } // Đẩy nhóm cuối cùng vào if (currentSubSlots.length > 0) { const ampm = currentGroupHour >= 12 ? 'PM' : 'AM'; const displayHour = currentGroupHour > 12 ? currentGroupHour - 12 : (currentGroupHour === 0 ? 12 : currentGroupHour); groups.push({ hourLabel: `${displayHour}:00 ${ampm}`, subSlots: currentSubSlots }); } return groups; } function minutesToTime(minutes) { const h = Math.floor(minutes / 60).toString().padStart(2, "0"); const m = (minutes % 60).toString().padStart(2, "0"); return `${h}:${m}`; } function getBookedCountsByDate(date) { const ss = SpreadsheetApp.openById(CONFIG.SHEET_ID); const sheet = ss.getSheetByName(CONFIG.SHEET_BOOKINGS); const data = sheet.getDataRange().getValues(); const counts = {}; for (let i = 1; i < data.length; i++) { const row = data[i]; let rowDateStr = row[2] instanceof Date ? Utilities.formatDate(row[2], "Australia/Melbourne", "yyyy-MM-dd") : String(row[2]).trim(); if (rowDateStr === date && row[13] === "confirmed") { const slotKey = row[3]; const numChildren = parseInt(row[9]) || 0; counts[slotKey] = (counts[slotKey] || 0) + numChildren; } } return counts; } function getCustomCapacity(date) { const ss = SpreadsheetApp.openById(CONFIG.SHEET_ID); const sheet = ss.getSheetByName(CONFIG.SHEET_CAPACITY); if (!sheet) return {}; const data = sheet.getDataRange().getValues(); const custom = {}; for (let i = 1; i < data.length; i++) { const row = data[i]; if (!row[0]) continue; let rowDateStr = row[0] instanceof Date ? Utilities.formatDate(row[0], "Australia/Melbourne", "yyyy-MM-dd") : String(row[0]).trim(); if (rowDateStr === date && row[1] && row[2]) { custom[row[1]] = parseInt(row[2]); } } return custom; } function generateId() { return "BK" + new Date().getTime().toString(36).toUpperCase(); } function sendConfirmationEmail(data, bookingId, slot, expiryStr) { const subject = "Booking Confirmed — My Little Dreamland (#" + bookingId + ")"; let formattedDate = data.date; try { const d = new Date(data.date + "T00:00:00"); formattedDate = d.toLocaleDateString("en-AU", { weekday: "long", day: "numeric", month: "long", year: "numeric" }); } catch (e) {} const htmlBody = `

⏳ Important: Spot Hold Notice

`; GmailApp.sendEmail(data.parentEmail, subject, "", { htmlBody, name: "My Little Dreamland", replyTo: "hello@mylittledreamland.com.au", }); } // ============================================================ // UTILS // ============================================================ function jsonResponse(data) { return ContentService.createTextOutput(JSON.stringify(data)).setMimeType(ContentService.MimeType.JSON); } function jsonError(message) { return ContentService.createTextOutput(JSON.stringify({ success: false, error: message })).setMimeType(ContentService.MimeType.JSON); } // ============================================================ // SETUP — Run once to create sheets // ============================================================ function setupSheets() { const ss = SpreadsheetApp.openById(CONFIG.SHEET_ID); let bookings = ss.getSheetByName(CONFIG.SHEET_BOOKINGS); if (!bookings) bookings = ss.insertSheet(CONFIG.SHEET_BOOKINGS); bookings.clearContents(); bookings.appendRow([ "ID", "Timestamp", "Date", "Slot Key", "Entry Time", "Exit Time", "Parent Name", "Email", "Phone", "Num Children", "Waiver Signed", "Photo Consent", "Notes", "Status", "Hold Expiry" ]); bookings.setFrozenRows(1); bookings.getRange("A1:O1").setBackground("#4A7259").setFontColor("#FFFFFF").setFontWeight("bold"); let capacity = ss.getSheetByName(CONFIG.SHEET_CAPACITY); if (!capacity) { capacity = ss.insertSheet(CONFIG.SHEET_CAPACITY); capacity.appendRow(["Date", "Slot Key", "Max Capacity", "Note"]); capacity.setFrozenRows(1); capacity.getRange("A1:D1").setBackground("#34A853").setFontColor("#FFFFFF").setFontWeight("bold"); } SpreadsheetApp.getUi().alert("Setup complete!"); }
